        throw new Error('Password must be at least 6 characters');
      }
    }

    // API calls with logging
    if (isLogin) {

      const response = await axiosInstance.post('/auth/login', { ... });

      // ... registration with similar structure

const ChatPage = () => {
  const messageRefreshInterval = useRef(null);

  const fetchMessages = async () => {
    try {
      console.log('[ChatPage] Fetching messages for match:', matchId);
      const response = await axiosInstance.get(`/messages/${matchId}`);
      console.log('[ChatPage] Messages loaded:', response.data);
      setMessages(response.data || []);
      setError('');
    } catch (err) {
      console.error('[ChatPage] Error fetching messages:', err);
      // ... error handling
    }
  };

  useEffect(() => {
    const initializeChat = async () => {
      try {
        setLoading(true);
        await fetchMessages();
        setLoading(false);

        // Poll for new messages every 2 seconds
        messageRefreshInterval.current = setInterval(fetchMessages, 2000);
      } catch (err) {
        // ...
      }
    };

    initializeChat();

    return () => {
      if (messageRefreshInterval.current) {
        clearInterval(messageRefreshInterval.current);
      }
    };
  }, [matchId]);

  const handleSendMessage = async (e) => {
    e.preventDefault();
    if (!newMessage.trim() || sending) return;

    try {
      setSending(true);
      console.log('[ChatPage] Sending message:', newMessage);
      
      const response = await axiosInstance.post('/messages', {
        matchId: matchId,  // Correct field
        content: newMessage
      });

      console.log('[ChatPage] Message sent:', response.data);
      setNewMessage('');
      
      // Fetch updated messages immediately after sending
      await fetchMessages();
    } catch (err) {
      console.error('[ChatPage] Error sending message:', err);
      setError(err.response?.data?.error || 'Failed to send message');
    } finally {
      setSending(false);
    }
  };
};

  • Line 71: Use matchId instead of recipientId
  • Line 74: Changed endpoint from /users to /messages
  • Lines 101-114: Token decoding for current user identification
  • Line 140: Fixed message timestamp field access
  • Added proper cleanup in useEffect
  • Complete component rewrite (90+ lines changed)
